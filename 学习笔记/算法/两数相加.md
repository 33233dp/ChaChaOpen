[[动态内存管理]]
``` cpp
struct ListNode {
	int val;
	ListNode* next;
	ListNode(int x) : val(x), next(nullptr) {}
};

class Solution {
public:
    ListNode* addTwoNumbers(ListNode* l1, ListNode* l2) {
	    //head和tail是新链表的头指针和尾指针
        ListNode *head = nullptr;
        ListNode *tail = nullptr;
        int carry = 0;
        while (l1 || l2) {
	        //使用三木运算避免访问空指针
            int n1 = l1 ? l1->val: 0;
            int n2 = l2 ? l2->val: 0;
            //carry是上一次相加旳进位（竖式加法）
            int sum = n1 + n2 + carry;
            if (!head) {//当Head为空的时候即第一次做加法
	            //开辟桟空间并返回地址，然后初始化节点存储的数值
	            //对于第一次，链表只有一个节点，因此头指针和尾指针都指向同一个节点。
                head = tail = new ListNode(sum % 10);
            } else {//head只指向地一个节点，只有tail需要更新，始终指向最后一个节点。
	            //尾插法：在末尾插入节点并初始化节点存储的值
                tail->next = new ListNode(sum % 10);
                //更新节点位置
                tail = tail->next;
            }
            carry = sum / 10;
            if (l1) {//如果没有还没有遍历完就继续移动
                l1 = l1->next;
            }
            if (l2) {
                l2 = l2->next;
            }
        }
        //遍历完之后，还有进位就再插入一个节点并初始化。
        if (carry > 0) {
            tail->next = new ListNode(carry);
        }
        return head;
    }
};
```
**核心思路**
	新链表的每一个节点都存放“两数和”的对应一位（个，十，百）。
	创建新链表的头指针和尾指针。
	每次循环都将当前两个节点的值加起来sum = n1 + n2，然后再加上上一次的进位sum / 10得到sum。
	插入新节点，然后初始化为sum % 10,因为每一位只能放下0-9的数字（竖式加法）。
	遍历完之后，检查进位sum / 10是否大于0,是的话就再在尾部插入一个节点，初始化为这个进位的值。
